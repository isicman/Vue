# 【题目 1】python调用api创建Domain

在自行搭建的 OpenStack私有云平台或提供的 all-in-one 平台上。在 controller 节点的/root目录下创建 create_domain.py 文件，编写 python 代码对接 OpenStack API，完成域的创建。要 求在 OpenStack 私有云平台中创建域名 chinaskill-demo，描述为“chinaskill_domain”。执行完代码 要求输出“域创建成功”。根据上述要求编写 python 代码，完成后，将 controller 节点的 IP 地址，用户名和密码提交。

[root@controller python]# cat create_domain.py  

import requests
import json
headers={}
headers['Content-Type']='application/json'
headers['X-Auth-Token']=''

body={
    "domain": {
        "description": "chinaskill_domain",
        "enabled": True,
        "name": "chinaskill-demo"
    }
}

a=requests.post(url='http://controller:35357//v3/domains',data=json.dumps(body),headers=headers)
print(a.text)
print("域创建成功")

# 【题目 6】python调用api创建镜像

在自行搭建的 OpenStack私有云平台或提供的 all-in-one 平台上。在 controller 节点的/root 目录下创建 create_image.py 文件，编写 python 代码对接 OpenStack API，完成镜像的上传。 要求在 OpenStack 私有云平台中上传镜像 cirros-0.3.4-x86_64-disk.img，名字为 pvm_image， disk_format 为 qcow2，container_format 为 bare。执行完代码要求输出“镜像创建成功，id 为:xxxxxx”。根据上述要求编写 python 代码，完成后，将 controller 节点的 IP 地址，用户 名和密码提交。

[root@controller python]# cat create_image.py 

import requests
import json 
headers={}
headers['Content-Type']='application/json'
headers['Content-Type']='application/octet-stream'
headers['X-Auth-Token']=''

body={
                "container_format": "bare",
                "disk_format": "qcow2",
                "name": "pvm_image",
}

request = requests.post('http://172.25.253.3:9292/v2/images', headers=headers, json=body).json() 
print(requests.put(f"http://172.25.253.3:9292{request['file']}", headers=headers, data=open("cirros-0.3.4-x86_64-disk.img", 'rb')).status_code)
a=requests.get('http://172.25.253.3:9292/v2/images', headers=headers, data=json.dumps(body))

在自行搭建的 OpenStack私有云平台或提供的 all-in-one 平台上。在 controller 节点的/root 目录下创建 create_vm.py 文件，编写 python 代码对接 OpenStack API，完成云主机的创建。 要求使用 pvm_image、pvm_flavor、pvm_intsubnet 创建 1 台云主机 pvm1（如果存在同名虚 拟主机，代码中需先进行删除操作）。执行完代码要求输出“创建云主机成功”。根据上述 要求编写 python 代码，完成后，将 controller 节点的 IP 地址，用户名和密码提交。

[root@controller python]# cat create_vm.py 

import requests
import json
headers = {}
headers["Content-Type"] = "application/json"
headers["X-Auth-Token"] = ""
body = {
    "server" : {
        "name" : "pvm1",
            #    //虚拟机名称
        "imageRef" : "06abde99-d78b-4749-bfb5-351a582b2d6e",
            #    //镜像UUID
        "flavorRef" : "fc8264e9-932d-4238-91d6-035f3ee00463",
            #    //实例类型UUID
        "networks" : [{"uuid" : "54cd4ead-8f46-48d1-85bf-5443178a5aa0"}],
            #    //配置选择网络
        "security_groups": [{"name": "pvm_sec"}]
            #    //配置安全组
    }
}

a = requests.post(url="http://10.11.121.75:8774/v2.1/servers",data=json.dumps(body),headers=headers)
print(a.text)
print("云主机建成功")

# 【题目 11】python调用api创建Router

在controller节点的/root目录下按要求编写Python程序create_route.py文件，对接 openstack api。完成路由器的创建。要求：

（1）通过获取openstack平台外网ID；

（2）创建路由pvm_router；

（3）添加内网的网卡接口；

（4）添加内网子网接口到路由，实现内外网网络互通（如果存在同名路由，代码中需先进行删除操作）。要求输出"路由器创建成功"和接口连接成功对应的路由名称、id和详细信息。

[root@controller python]# cat create_router.py

import requests
import json
headers={}
headers['Content-Type']='application/json'
headers['X-Auth-Token']=''
body={
        'router': {
                'name': 'pvm_router',
                'external_gateway_info': {
                        'network_id': 'fd09d6c6-11df-4897-aca1-3ed4ea4efe48'				#外网卡的ID
                }
        }
}
a=requests.post(url='http://172.25.253.3:9696/v2.0/routers',data=json.dumps(body),headers=headers)
print(a.text)
print("路由器外网接口创建成功")

info = json.loads(a.text)
id=info['router']['id']
print("路由器的id"+id)

twobody={
                'subnet_id': '19b9884d-f5b8-4b59-b60b-1aa18cb9fb76'							#内网子网的ID
}

#/v2.0/routers/{project_id}/add_router_interface
b=requests.put(url="http://172.25.253.3:9696/v2.0/routers/"+id+"/add_router_interface",data=json.dumps(twobody),headers=headers)

print('路由器内网接口成功')

# 【题目 13】python调用api挂载卷

在 controller 节点的/root 目录下按要求编写 Python 程序 attach_lvm.py 文件，对接openstack api，扩展刚刚创建的云硬盘大小 为50G，要求将 pvm_cdisk 卷附加到云主机 pvm1。输出云硬盘挂载成功云主机名称、ID 和详细信息。

[root@controller python]# attach_lvm.py  

import requests
import json

headers={}
headers['Content-Type']='application/json'
headers['X-Auth-Token']=''

body={
    'os-attach':{
        'instance_uuid':'0a3e7257-3752-422d-b219-ef2ee7c8bd2a',		#server的uuid
        'mountpoint':'/dev/vdb'
    }
}

a=requests.post(url='http://172.129.33.11:8776/v3/d01c7adaf105421d927f57f3fcac3e7c/volumes/4b78cd7f-ebae-4bcb-8691-5836ecb3d871/action',data=json.dumps(body),headers=headers)
print(a.text)
print("云硬盘挂载成功")

# 【题目 14】python调用api创建Stack

使用云平台搭建heat环境，在controller节点的/root目录下按要求编写Python程序create_stack.py文件，对接 openstack api，完成云主机类型的创建，实现创建主机类型名称为m1.flavor：ID为123456、内存1024M、硬盘20G、vcpu 2个（如果存在同名堆栈，代码中需先进行删除操作）。输出堆栈创建成功堆栈名、id和详细信息。

[root@controller python]# create_stack.py  

import requests
import json

headers={}
headers['Content-Type']='application/json'
headers['X-Auth-Token']=''

body={
    'files':{},
    'stack_name':'teststack',
    'template':{
        'heat_template_version':'2013-05-23',
        'resources':{
            'hello_world':{
                'type':'OS::Nova::Flavor',
                'properties':{
                    'name':'m1.flavor',
                    'disk':'20',
                    'ram':'1024',
                    'vcpus':'2',
                    'flavorid':'123456'
                    }
                }
            }
        }
    
}

a=requests.post(url='http://172.129.33.11:8004/v1/d01c7adaf105421d927f57f3fcac3e7c/stacks',data=json.dumps(body),headers=headers)
print(a.text)
print('堆栈创建成功')
